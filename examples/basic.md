# أمثلة: أساسيات لغة ص

## مرحباً بالعالم
```sad
اطبع_سطر("مرحباً بالعالم!")
```

## متغيرات وحساب
```sad
متغير اسم = "أحمد"
رقم عمر = 25
اطبع_سطر(م"{اسم} عمره {عمر}")     # نص منسّق
```

## شرط
```sad
دالة صنّف(رقم درجة)
    إذا (درجة >= 90)
        ارجع "ممتاز"
    والا_اذا (درجة >= 75)
        ارجع "جيد جدًا"
    وإلا
        ارجع "مقبول"
    نهاية
نهاية
اطبع_سطر(صنّف(82))                # جيد جدًا
```

## حلقات
```sad
متغير مجموع = 0
لكل ع في 1..10
    مجموع = مجموع + ع
نهاية
اطبع_سطر(مجموع)                   # 45
```

## دالة ولامدا
```sad
دالة رقم جمع(رقم أ، رقم ب)
    ارجع أ + ب
نهاية

متغير ضعف = لامدا(س) => س * 2
اطبع_سطر(جمع(3، 4))               # 7
اطبع_سطر(ضعف(10))                # 20
```

## مصفوفة وخريطة
```sad
متغير أرقام = [5، 2، 8، 1]
اطبع_سطر(فرز(أرقام))                          # [1، 2، 5، 8]
اطبع_سطر(تخطيط(أرقام، لامدا(ن) => ن * ن))     # المربّعات

متغير طالب = {"اسم": "سارة"، "معدل": 95}
اطبع_سطر(طالب["اسم"])
طالب["نشط"] = صحيح
```

## مطابقة الأنماط
```sad
دالة وصف(رقم ن)
    طابق (ن)
        عندما 0:        ارجع "صفر"
        عندما 1..9:     ارجع "خانة واحدة"
        عندما _:        ارجع "كبير"
    نهاية
نهاية
اطبع_سطر(وصف(7))                  # خانة واحدة
```

## صنف بسيط
```sad
صنف عدّاد
    باني()
        هذا.قيمة = 0
    نهاية
    دالة زد()
        هذا.قيمة = هذا.قيمة + 1
    نهاية
نهاية

متغير ع = جديد عدّاد()
ع.زد()
ع.زد()
اطبع_سطر(ع.قيمة)                  # 2
```
