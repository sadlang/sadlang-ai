# أمثلة: نظام الأنواع

## استنتاج وتصريح
```sad
متغير أ = 5            # رقم (مستنتج)
عشري ب = 3.5           # عشري صريح
نص ج = "نص"
منطقي د = صحيح
اطبع_سطر(نوع(أ))      # رقم
اطبع_سطر(نوع(ج))      # نص
```

## صنف مع وراثة وتعدد أشكال
```sad
صنف حيوان
    باني(الاسم)
        هذا.اسم = الاسم
    نهاية
    دالة صوت()
        ارجع "..."
    نهاية
    دالة وصف()
        ارجع هذا.اسم + " يقول " + هذا.صوت()
    نهاية
نهاية

صنف قطة يرث حيوان
    دالة صوت()             # تجاوز
        ارجع "مياو"
    نهاية
نهاية

متغير ق = جديد قطة("لولو")
اطبع_سطر(ق.وصف())          # لولو يقول مياو
```

## الأعضاء الساكنة
```sad
صنف مستخدم
    ساكن رقم العدد = 0
    باني(الاسم)
        هذا.اسم = الاسم
        مستخدم.العدد = مستخدم.العدد + 1   # عبر اسم الصنف
    نهاية
    ساكن دالة رقم المجموع()
        ارجع مستخدم.العدد
    نهاية
نهاية

متغير م1 = جديد مستخدم("أ")
متغير م2 = جديد مستخدم("ب")
اطبع_سطر(مستخدم.المجموع())   # 2
```

## السمات (Traits)
```sad
سمة قابل_للطباعة
    دالة اطبع()
نهاية

صنف كتاب
    باني(العنوان)
        هذا.العنوان = العنوان
    نهاية
نهاية

نفّذ قابل_للطباعة لـ كتاب
    دالة اطبع()
        اطبع_سطر("كتاب: " + هذا.العنوان)
    نهاية
نهاية

متغير ك = جديد كتاب("النحو")
ك.اطبع()                    # كتاب: النحو
```

## قيود الأنواع (Where)
```sad
دالة مضاعفة(أ)
    حيث ت: رقمي
    ارجع أ * 2
نهاية

اطبع_سطر(مضاعفة(5))        # 10
# مضاعفة("نص")             # RuntimeError: فشل قيد النوع

دالة أكبر(أ، ب)
    حيث ت: قابل_للمقارنة    # أرقام أو نصوص
    إذا (أ > ب) ارجع أ نهاية
    ارجع ب
نهاية
اطبع_سطر(أكبر(3، 9))       # 9
اطبع_سطر(أكبر("أ"، "ب"))   # ب
```

## بنية (حقول فقط) + دالة مصنّعة
```sad
بنية نقطة
    رقم س = 0
    رقم ص = 0
نهاية

دالة نقطة_جديدة(س، ص)
    متغير ن = جديد نقطة()
    ن.س = س
    ن.ص = ص
    ارجع ن
نهاية

متغير ن = نقطة_جديدة(3، 5)
اطبع_سطر(ن.س)             # 3
```
