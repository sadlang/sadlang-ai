# أمثلة: الملكية والأعمار

## التمرير بالقيمة مقابل المرجع
```sad
# تُمرَّر بالقيمة — التعديل يضيع
دالة جرّب_تعديل(عدد)
    عدد = 999
نهاية

متغير س = 10
جرّب_تعديل(س)
اطبع_سطر(س)              # 10 — لم يتغيّر

# بالمرجع باستخدام &
دالة زيادة(&عدد، مقدار)
    عدد = عدد + مقدار
نهاية
زيادة(&س، 5)
اطبع_سطر(س)              # 15
```

## المصفوفات تُمرَّر بالقيمة — استعمل صنفًا للحالة القابلة للتعديل
```sad
# ❌ لن يعمل
دالة بدّل(مصفوفة، ع1، ع2)
    متغير مؤقت = مصفوفة[ع1]
    مصفوفة[ع1] = مصفوفة[ع2]
    مصفوفة[ع2] = مؤقت
نهاية

# ✅ غلّف في صنف
صنف كومة
    باني()
        هذا._بيانات = []
    نهاية
    دالة ادفع(عنصر)
        هذا._بيانات.اضف(عنصر)
    نهاية
    دالة بدّل(ع1، ع2)
        متغير مؤقت = هذا._بيانات[ع1]
        هذا._بيانات[ع1] = هذا._بيانات[ع2]
        هذا._بيانات[ع2] = مؤقت
    نهاية
نهاية
```

## تعليقات العمر
```sad
دالة أطول<'أ>(س، ص)
    إذا (طول(س) > طول(ص))
        ارجع س
    وإلا
        ارجع ص
    نهاية
نهاية
اطبع_سطر(أطول("قصير"، "أطول بكثير"))

# عدة أعمار
دالة دمج<'أ، 'ب>(أول، ثاني)
    ارجع أول + " و " + ثاني
نهاية

# عمر ثابت
دالة شعار<'ثابت>()
    ارجع "لغة ص"
نهاية
```

## التأجيل المضمون
```sad
دالة افتح_ملف()
    أجّل اطبع_سطر("إغلاق المورد")
    اطبع_سطر("استخدام المورد")
    # الخرج: استخدام المورد ← إغلاق المورد
نهاية

# LIFO
دالة ترتيب()
    أجّل اطبع_سطر("أول")
    أجّل اطبع_سطر("ثاني")
    أجّل اطبع_سطر("ثالث")
    # الخرج: ثالث ← ثاني ← أول
نهاية
```
